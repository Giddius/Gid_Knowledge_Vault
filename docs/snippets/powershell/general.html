Get all local Git Repo Paths

Set-Location -Path D:\; Get-ChildItem . -Attributes Directory+Hidden -ErrorAction SilentlyContinue -Filter ".git" -Recurse | % { Write-Host $_.FullName -replace"\\.git" }
